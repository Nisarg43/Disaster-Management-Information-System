Final Queries



-- 1. Count disasters per severity + avg duration
SELECT Severity_Level, COUNT(*) AS Total,
       ROUND(AVG(DATEDIFF(End_Date, Start_Date)), 1) AS Avg_Days
FROM Disaster
GROUP BY Severity_Level
ORDER BY Total DESC;

-- 2. Worst regions by total deaths 
SELECT r.District, d.Description, aa.Deceased_People, aa.Damage_Level
FROM Affected_Area aa
INNER JOIN Disaster d  ON aa.Disaster_ID = d.Disaster_ID
INNER JOIN Region   r  ON aa.Region_ID   = r.Region_ID
ORDER BY aa.Deceased_People DESC;

-- 3. Rescue efficiency ratio per disaster 
SELECT d.Description,
       SUM(aa.Rescued_People) AS Rescued,
       SUM(aa.Injured_People) AS Injured,
       ROUND(SUM(aa.Rescued_People) / NULLIF(SUM(aa.Injured_People),0), 2) AS Ratio
FROM Affected_Area aa
INNER JOIN Disaster d ON aa.Disaster_ID = d.Disaster_ID
GROUP BY d.Description
ORDER BY Ratio DESC;

-- 4. Volunteer skill count per organisation 
SELECT o.Name AS Organisation, v.Skill, COUNT(*) AS Volunteers
FROM Volunteer v
INNER JOIN Organisation o ON v.Organisation_ID = o.Organisation_ID
GROUP BY o.Name, v.Skill
ORDER BY o.Name, Volunteers DESC;

-- 5. Fund balance — allocated vs actually spent 
SELECT f.Fund_ID, f.Purpose_Descriptor, f.Amount AS Allocated,
       COALESCE(SUM(e.Amount_Spent), 0) AS Spent,
       f.Amount - COALESCE(SUM(e.Amount_Spent), 0) AS Balance
FROM Fund f
LEFT JOIN Expenditure e ON f.Fund_ID = e.Fund_ID
GROUP BY f.Fund_ID, f.Purpose_Descriptor, f.Amount;

-- 6. Org summary — volunteers + shelters + supplies 
SELECT o.Name, o.Organisation_Type,
       COUNT(DISTINCT v.Volunteer_ID) AS Volunteers,
       COUNT(DISTINCT s.Shelter_ID)   AS Shelters,
       COUNT(DISTINCT rs.Supply_ID)   AS Supplies
FROM Organisation o
LEFT JOIN Volunteer     v  ON v.Organisation_ID = o.Organisation_ID
LEFT JOIN Shelter       s  ON s.Organisation_ID = o.Organisation_ID
LEFT JOIN Relief_Supply rs ON rs.Org_ID         = o.Organisation_ID
GROUP BY o.Name, o.Organisation_Type;

-- 7. Shelter capacity deficit vs rescued people 
SELECT r.District,
       SUM(s.Capacity)        AS Shelter_Cap,
       SUM(aa.Rescued_People) AS Rescued
FROM Region r
LEFT JOIN Shelter       s  ON r.Region_ID = s.Region_ID
LEFT JOIN Affected_Area aa ON r.Region_ID = aa.Region_ID
GROUP BY r.District
HAVING SUM(aa.Rescued_People) > COALESCE(SUM(s.Capacity), 0);

-- 8. Occupancy % per district — only >70% 
SELECT r.District,
       SUM(s.Current_Occupancy) AS Occupied,
       SUM(s.Capacity)          AS Total,
       ROUND(SUM(s.Current_Occupancy)/SUM(s.Capacity)*100, 1) AS Pct
FROM Shelter s
INNER JOIN Region r ON s.Region_ID = r.Region_ID
GROUP BY r.District
HAVING Pct > 70
ORDER BY Pct DESC;

-- 9. Supply categories with total units > 500 
SELECT Category,
       COUNT(*)         AS Line_Items,
       SUM(No_of_Units) AS Total_Units
FROM Relief_Supply
GROUP BY Category
HAVING Total_Units > 500
ORDER BY Total_Units DESC;

-- 10. Districts with more than 100 total injured 
SELECT r.District, SUM(aa.Injured_People) AS Total_Injured
FROM Affected_Area aa
INNER JOIN Region r ON aa.Region_ID = r.Region_ID
GROUP BY r.District
HAVING Total_Injured > 100
ORDER BY Total_Injured DESC;

-- 11. Orgs assigned to Critical damage zones 
SELECT DISTINCT o.Name, o.Organisation_Type
FROM Organisation o
INNER JOIN Assigned a ON a.Organisation_ID = o.Organisation_ID
WHERE a.Region_ID IN (
    SELECT Region_ID FROM Affected_Area WHERE Damage_Level = 'Critical'
);

-- 12. Disasters with above-average total casualties 
SELECT d.Description,
       SUM(aa.Deceased_People + aa.Injured_People) AS Total_Hit
FROM Affected_Area aa
INNER JOIN Disaster d ON aa.Disaster_ID = d.Disaster_ID
GROUP BY d.Description
HAVING Total_Hit > (
    SELECT AVG(Deceased_People + Injured_People) FROM Affected_Area
);

-- 13. Funds with amount above the overall average 
SELECT Fund_ID, Fund_Type, Purpose_Descriptor, Amount
FROM Fund
WHERE Amount > (SELECT AVG(Amount) FROM Fund)
ORDER BY Amount DESC;

-- 14. Classify rescue performance using CASE WHEN
SELECT d.Description,
       SUM(aa.Rescued_People) AS Rescued,
       CASE
           WHEN SUM(aa.Injured_People) = 0 THEN 'No injury data'
           WHEN SUM(aa.Rescued_People)/SUM(aa.Injured_People) >= 1 THEN 'Good'
           WHEN SUM(aa.Rescued_People)/SUM(aa.Injured_People) >= 0.5 THEN 'Moderate'
           ELSE 'Poor'
       END AS Rescue_Rating
FROM Affected_Area aa
INNER JOIN Disaster d ON aa.Disaster_ID = d.Disaster_ID
GROUP BY d.Description;

-- 15. Classify shelter status using CASE WHEN
SELECT s.Shelter_ID, r.District,
       s.Capacity, s.Current_Occupancy,
       CASE
           WHEN s.Current_Occupancy = 0              THEN 'Unused'
           WHEN s.Current_Occupancy = s.Capacity     THEN 'Full'
           WHEN s.Current_Occupancy/s.Capacity > 0.8 THEN 'Near Full'
           ELSE 'Available'
       END AS Status
FROM Shelter s
INNER JOIN Region r ON s.Region_ID = r.Region_ID;

-- 16. Organisations with NO volunteers at all (NOT EXISTS)
SELECT o.Name, o.Organisation_Type
FROM Organisation o
WHERE NOT EXISTS (
    SELECT 1 FROM Volunteer v
    WHERE v.Organisation_ID = o.Organisation_ID
);

-- 17. Orgs appearing in BOTH supply and expenditure 
SELECT o.Name
FROM Organisation o
WHERE o.Organisation_ID IN (SELECT Org_ID FROM Relief_Supply)
  AND o.Organisation_ID IN (SELECT Org_ID FROM Expenditure);

-- 18. All Government and International NGO orgs (UNION)
SELECT Name, Organisation_Type FROM Organisation WHERE Organisation_Type = 'Government'
UNION
SELECT Name, Organisation_Type FROM Organisation WHERE Organisation_Type = 'International NGO'
ORDER BY Organisation_Type, Name;

-- 19. Volunteer → Assigned → Disaster → Region 
SELECT v.Name, v.Skill, d.Description AS Disaster, r.District
FROM Assigned a
INNER JOIN Volunteer v ON a.Email_ID       = v.Email_ID
INNER JOIN Disaster  d ON a.Disaster_ID    = d.Disaster_ID
INNER JOIN Region    r ON a.Region_ID      = r.Region_ID
ORDER BY d.Description, v.Skill;

-- 20. Expenditure cross-tab: org type vs fund type
SELECT o.Organisation_Type, f.Fund_Type,
       COUNT(*) AS Transactions, SUM(e.Amount_Spent) AS Total_Spent
FROM Expenditure e
INNER JOIN Organisation o ON e.Org_ID  = o.Organisation_ID
INNER JOIN Fund         f ON e.Fund_ID = f.Fund_ID
GROUP BY o.Organisation_Type, f.Fund_Type
ORDER BY o.Organisation_Type, Total_Spent DESC;

-- 21. Monthly spending trend in 2026
SELECT DATE_FORMAT(Date, '%Y-%m') AS Month,
       COUNT(*)              AS Transactions,
       SUM(Amount_Spent)     AS Monthly_Spend,
       ROUND(AVG(Amount_Spent), 0) AS Avg_Per_Transaction
FROM Expenditure
WHERE YEAR(Date) = 2026
GROUP BY Month ORDER BY Month;

-- 22. Master audit — 5-table JOIN with CASE severity label
SELECT d.Description, r.District, o.Name AS Org,
       aa.Damage_Level, aa.Deceased_People,
       CASE d.Severity_Level
           WHEN 'Critical' THEN 'Immediate action'
           WHEN 'Severe'   THEN 'High priority'
           ELSE 'Monitor'
       END AS Response_Flag
FROM Affected_Area aa
INNER JOIN Disaster      d ON aa.Disaster_ID    = d.Disaster_ID
INNER JOIN Region        r ON aa.Region_ID      = r.Region_ID
INNER JOIN Assigned      a ON a.Disaster_ID     = d.Disaster_ID AND a.Region_ID = r.Region_ID
INNER JOIN Organisation  o ON a.Organisation_ID = o.Organisation_ID;

-- 23. Weighted human impact score per region
SELECT r.District,
       SUM(aa.Deceased_People * 5 +
           aa.Missing_People  * 3 +
           aa.Injured_People  * 1) AS Impact_Score
FROM Affected_Area aa
INNER JOIN Region r ON aa.Region_ID = r.Region_ID
GROUP BY r.District
ORDER BY Impact_Score DESC;

-- 24. Show total volunteers deployed per disaster and region
SELECT d.Description AS Disaster,
       r.District,
       COUNT(DISTINCT v.Email_ID) AS Total_Volunteers
FROM Assigned a
INNER JOIN Volunteer v ON a.Email_ID = v.Email_ID
INNER JOIN Disaster  d ON a.Disaster_ID = d.Disaster_ID
INNER JOIN Region    r ON a.Region_ID = r.Region_ID
GROUP BY d.Description, r.District
ORDER BY Total_Volunteers DESC;

-- 25. Find organisations handling maximum funds
SELECT o.Name,
       SUM(f.Amount) AS Total_Funds
FROM Organisation o
INNER JOIN Fund f ON o.Organisation_ID = f.Org_ID
GROUP BY o.Name
ORDER BY Total_Funds DESC
LIMIT 1;

-- 26. Find latest expenditure entry for each organisation
SELECT e.*
FROM Expenditure e
WHERE (e.Org_ID, e.Date) IN (
    SELECT Org_ID, MAX(Date)
    FROM Expenditure
    GROUP BY Org_ID
);
